POST /CreateActor //Создает актёра. При ответе 200 вернется структура с информацией об Актёре, иначе 500
{
    "actorname": "Иванов Иван", //Имя Актёра
    "sex": "Мужской", //Пол
    "birthdate": "1991-07-02", //Дата рождения
    "addfilms": [1, 2] //Массив из ИД фильмов, в которые нужно добавить Актёра. Не обязательный параметр
